-- Ejercicio 1.a
CREATE VIEW vista_pedidos_detalle AS
SELECT 
	pedido.id_pedido,
	clientes.razon_social,
	productos.nombre,
	detalle_pedido.cantidad,
	detalle_pedido.precio_venta,
	SUM(detalle_pedido.cantidad * detalle_pedido.precio_venta) AS subtotal
FROM detalle_pedido
JOIN pedidos ON pedidos.id_pedido = detalle_pedido.id_pedido
JOIN productos ON producto.id_producto = detalle_pedido.id_producto
JOIN clientes ON clientes.id_cliente = pedidos.id_cliente;

-- Ejercicio 1.b

CREATE VIEW vista_stock_bajo AS 
SELECT id_producto, nombre, categoria, stock_actual
FROM productos
WHERE stock_actual > 50;

-- Ejercicio 1.c
SELECT razon_social,
SUM(subtotal) AS total_facturado
FROM vista_detalle_pedido
ORDER BY total_facturado DESC;

-- Ejercicio 2.a
DELIMITER $$
CREATE TRIGGER tgr_actualizar_stock
AFTER INSERT ON detalle_pedido
FOR EACH ROW
BEGIN
	UPDATE productos
	SET stock_actual = stock_actual - NEW.cantidad
	WHERE id.producto = NEW.id_producto
END $$
DELIMITER ;

-- Ejercicio 2.b
DELIMITER $$
CREATE TRIGGER trg_auditoria_stock
AFTER UPDATE ON productos
FOR EACH ROW
BEGIN
	IF OLD.stock_actual != NEW.stock_actual THEN
	INSERT INTO auditoria_stock (id_producto, stock_anterior, stock_nuevo, fecha_cambio)
	VALUE(OLD.id_producto, OLD.stock_anterior, NEW.stock_nuevo, NOW());
END $$
DELIMITER ;
