CREATE DATABASE almacen_mayorista;
USE almacen_mayorista;
CREATE TABLE productos (    
id_producto INT PRIMARY KEY AUTO_INCREMENT,
nombre VARCHAR(100) NOT NULL,    
categoria VARCHAR(50) NOT NULL,    
precio_unitario DECIMAL(10,2) NOT NULL,    
stock_actual INT NOT NULL DEFAULT 0 );

CREATE TABLE clientes (    
id_cliente INT PRIMARY KEY AUTO_INCREMENT,
razon_social VARCHAR(100) NOT NULL,
cuit VARCHAR(15) NOT NULL UNIQUE,
saldo_cuenta DECIMAL(10,2) NOT NULL DEFAULT 0); 

CREATE TABLE pedidos (    
id_pedido INT PRIMARY KEY AUTO_INCREMENT,
id_cliente INT NOT NULL,
fecha_pedido DATE NOT NULL,
estado VARCHAR(20) NOT NULL DEFAULT 'PENDIENTE',
FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente) );

CREATE TABLE detalle_pedido (    
id_detalle INT PRIMARY KEY AUTO_INCREMENT,
id_pedido INT NOT NULL,
id_producto  INT NOT NULL,
cantidad INT NOT NULL,
precio_venta DECIMAL(10,2) NOT NULL,
FOREIGN KEY (id_pedido) REFERENCES pedidos(id_pedido),
FOREIGN KEY (id_producto) REFERENCES productos(id_producto) );

CREATE TABLE auditoria_stock (    
id_auditoria  INT PRIMARY KEY AUTO_INCREMENT,
id_producto   INT NOT NULL,
stock_anterior INT NOT NULL,
stock_nuevo INT NOT NULL,
fecha_cambio   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
FOREIGN KEY (id_producto) REFERENCES productos(id_producto) );

--1A)
CREATE VIEW vista_pedidos_detalle AS 
SELECT ped.id_pedido,
cli.razon_social,
pro.nombre,
det.cantidad,
det.precio_venta,
SUM(det.cantidad * det.precio_venta) AS 'subtotal'
FROM detalle_pedido det
INNER JOIN pedidos ped ON ped.id_pedido = det.id_pedido
INNER JOIN clientes cli ON cli.id_cliente = ped.id_cliente
INNER JOIN productos pro ON pro.id_producto = det.id_producto
GROUP BY ped.id_pedido, cli.razon_social, pro.nombre, det.cantidad, det.precio_venta;
--1B)
CREATE VIEW vista_stock_abajo AS
SELECT id_producto,
nombre,
categoria,stock_Actual FROM productos WHERE stock_actual < 50;

SELECT razon_social, SUM(subtotal) AS 'suma_subtotales' FROM vista_pedidos_detalle
