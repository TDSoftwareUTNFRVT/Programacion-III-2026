EJERCICIOS DE BASE DE DATOS TRIGGERS

TABLAS CON INSERCION DE DATOS:

CREATE TABLE clientes (
    id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE,
    saldo DECIMAL(12,2) DEFAULT 0.00,
    activo TINYINT(1) DEFAULT 1,
    fecha_registro DATETIME DEFAULT NOW()
);

CREATE TABLE productos (
    id_producto INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    precio DECIMAL(10,2) NOT NULL,
    stock INT DEFAULT 0,
    fecha_modificacion DATETIME
);

CREATE TABLE ventas (
    id_venta INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT NOT NULL,
    id_producto INT NOT NULL,
    cantidad INT NOT NULL,
    total DECIMAL(12,2),
    fecha DATETIME DEFAULT NOW(),
    FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente),
    FOREIGN KEY (id_producto) REFERENCES productos(id_producto)
);

CREATE TABLE auditoria_precios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_producto INT,
    precio_anterior DECIMAL(10,2),
    precio_nuevo DECIMAL(10,2),
    usuario VARCHAR(100),
    fecha_cambio DATETIME DEFAULT NOW()
);

CREATE TABLE log_clientes_eliminados (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT,
    nombre VARCHAR(100),
    email VARCHAR(150),
    fecha_eliminacion DATETIME DEFAULT NOW()
);

DELIMITER $$

CREATE TRIGGER validar_stock_antes_venta
BEFORE INSERT ON ventas
FOR EACH ROW
BEGIN
    DECLARE stock_disponible INT;

    Consultar el stock disponible del producto:
    SELECT stock INTO stock_disponible
    FROM productos
    WHERE id_producto = NEW.id_producto;

    Validar que exista stock suficiente:
    IF stock_disponible < NEW.cantidad THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Aviso: No hay stock suficiente para concretar la venta';
    END IF;
END$$

DELIMITER ;

DELIMITER $$

CREATE TRIGGER descontar_saldo_cliente
AFTER INSERT ON ventas
FOR EACH ROW
BEGIN
    Restar el total de la venta al saldo del cliente:
    UPDATE clientes
    SET saldo = saldo - NEW.total
    WHERE id_cliente = NEW.id_cliente;
END$$

DELIMITER ;

DELIMITER $$

CREATE TRIGGER controlar_email_cliente
BEFORE UPDATE ON clientes
FOR EACH ROW
BEGIN
    DECLARE email_repetido INT;

    Verificar solo si el email cambio:
    IF NEW.email <> OLD.email THEN

        Pasar el nuevo email a minusculas:
        SET NEW.email = LOWER(NEW.email);

        Contar si existe otro cliente con el mismo email:
        SELECT COUNT(*) INTO email_repetido
        FROM clientes
        WHERE email = NEW.email
        AND id_cliente <> NEW.id_cliente;

        Lanzar error si esta duplicado:
        IF email_repetido > 0 THEN
            SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'Aviso: Ese email ya pertenece a otro cliente';
        END IF;

    END IF;
END$$

DELIMITER ;

DELIMITER $$

CREATE TRIGGER reponer_stock_al_borrar_venta
AFTER DELETE ON ventas
FOR EACH ROW
BEGIN
    Devolver las unidades al stock del producto:
    UPDATE productos
    SET stock = stock + OLD.cantidad,
        fecha_modificacion = NOW()
    WHERE id_producto = OLD.id_producto;
END$$

DELIMITER ;
